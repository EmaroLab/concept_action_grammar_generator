78  public TermOccurenceCollector getAllTermOccurence( Long startingId, String termStr) {
79  // create an empty output
80  TermOccurenceCollector out = new TermOccurenceCollector( termStr);
81  // check if the term is in grammar (in order to populated SkippedWorld consistently)
82  // it may be not efficient !!!!!!! since requires to iterate two times over expressionTermMap.
83  Boolean grammarContainsTerm = false;
84  for( SemanticExpressionNodeBase node : expressionTermMap.values()){
85  ExpressionTermData termData = ( ExpressionTermData) node.getSyntaxData();
86  if( ! termData.getInstance2()){ // if it is not a rule declaration
87  String termDefinition = termData.getInstance();
88  if( termDefinition.equalsIgnoreCase( termStr))
89  grammarContainsTerm = true;
90  }
91  if( grammarContainsTerm)
92  break;
93  }
94  if( ! grammarContainsTerm)
95  return null;

97 
98  // for all the terms of the expression tree
99  for( SemanticExpressionNodeBase node : expressionTermMap.values()){
100  // get ordering node id
101  Long termId = node.getId();
102  // check if unfeasible
103  if( termId <= startingId)
104  out.getUnfeasibleTermComposition().add( termId);
105  else { // if feasible
106  ExpressionTermData termData = ( ExpressionTermData) node.getSyntaxData();
107  if( ! termData.getInstance2()){ // if it is not a rule declaration
108  String termDefinition = termData.getInstance();

111  }
112  }
113  }
114  }
115  // order by increasing id
116  Collections.sort( out.getUnfeasibleTermComposition());
117  Collections.sort( out.getFeasibleTermComposition());
118  return out;
