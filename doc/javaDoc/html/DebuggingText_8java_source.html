451  private synchronized String pad( String str, boolean left, int size){ // to make the string of the same length by adding/removing char on left or right side
452  if( str.length() == size)
453  return str;
454  if( str.length() < size){
455  if( left)
456  return String.format("%1$" + size + "s", str);
457  else return String.format("%1$-" + size + "s", str);
458  } else {
459  if( left)
460  return SYST_PROPERTY_CALLER_CUT_SYMBOL + str.substring( str.length() - size + SYST_PROPERTY_CALLER_CUT_SYMBOL.length());
461  else return str.substring( 0, size - SYST_PROPERTY_CALLER_CUT_SYMBOL.length()) + SYST_PROPERTY_CALLER_CUT_SYMBOL;
462  }
