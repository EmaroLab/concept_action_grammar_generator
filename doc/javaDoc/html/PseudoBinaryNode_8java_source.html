567  public N buildTree(){
568  // if order is 1 returns just the given leaf. (order = 0 managed through polymorphism)
569  if( getOrder() == 1)
570  return getLeafNode( 0);
571  // otherwise create binary tree of the specified order
572  N root = getOperatorNode( 0);//getOperator( useAndOperator); // initialise the root to be returned
573  N actualNode = root; // initialise the current node used for binary tree creation
574  if( getOrder() == 0){
575  actualNode.addChild( getLeafNode( 0)); // add the child of optional node (only one child no left or right)
576  } else { // order > 1
577  for( int k = 0; k < getOrder() - 1; k++) // for the number of operation node to be created - 1 (since root is already an operator).
578  if( this.useLeftPriority)
579  actualNode = buildLeftTree( actualNode, k); // add first leaf and than operator
580  else actualNode = buildRightTree( actualNode, k); // add first operator and than leaf
581  actualNode.addChild( getLeafNode( getOrder() - 1)); // add the last right leaf
582  }
583  return root; // returns the root of the created binary tree,
584  }
