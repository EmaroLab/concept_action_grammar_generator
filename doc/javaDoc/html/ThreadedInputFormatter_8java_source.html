201  private List< Long> combineRecorsive( Integer idx, List< Long> inputIds){ // recursive call and term id combination list creation
202  if( isRunning()){
203  if( idx < userInput.size()){
204  if( ! isInSkipped( idx)){
205  // get the last analyzed id. If no one use 0
206  Long startingId;
207  if( inputIds.isEmpty()) // it should never be null

210  // search for the term occurrences on the term map (the grammar object does it!!)
211  GrammarBase<? extends SemanticExpressionTreeBase>.TermOccurenceCollector occurrences = getTermOccurence( startingId, idx);
212  if( occurrences == null){
213  SkippedWord skipWord = new SkippedWord( userInput, idx); // the index may be wrong in case of "^" usage
214  skipped.add( skipWord);
215  inputIds = combineRecorsive( idx + 1, inputIds);
216  } else {
217  // notify unfeasible solution and continue with feasible ones
218  storeUnfeasibleSolution( inputIds, occurrences.getUnfeasibleTermComposition());

220  // if no occurrences skip the world given by the user (userInput.get( idx)) since it is not on the grammar
221  if( ! termOccurenceId.isEmpty()){
222  for( Long id : termOccurenceId){ // accept this sub solution and continue searching iteratively
223  inputIds.add( id);
224  inputIds = combineRecorsive( idx + 1, inputIds);
225  inputIds = copyListIterative( inputIds);
226  }
227  }
228  }
229  } else {

231  inputIds = combineRecorsive( idx + 1, inputIds);
232  }
233  } // do not use ELSE statement since idx and inputIds can change during iteration.
234  if( idx >= inputIds.size()) // Check the condition again instead
235  storeFeasibleSolution( inputIds);
236  } else if( ! alreadyLoggedFlag){
237  logWarning( "++ fact searcher stopped !!!");
238  alreadyLoggedFlag = true;
