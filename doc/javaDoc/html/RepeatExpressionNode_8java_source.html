208  // convert a repeat expression into an equivalent translation composed by a Boolean Expression Tree.
209  private ExpressionNode<?> getTranslatedExpression( Integer min, Integer max, RepeatExpressionNode repeatNode){
210  // create all the binary trees to represents that the expression is repeated K times
211  // ex: E(0) = !optional(E) ; E(1)=E ; E(2)=E&E ; E(3)=E&E&E ; E(4)=E&E&E&E ; .... E(i) with i in [min, max]
212  ExpressionNode<?> originalExpression = (ExpressionNode<?>) repeatNode.getChild( 0); // (see assignMaxNumberOfChildren())
213  List< ExpressionNode<?>> repetitionExpressions = new ArrayList< ExpressionNode<?>>();
214  for( int i = min; i <= max; i++){
215  ExpressionNode<?> repetition;
216  if( i == 0) // create the !optional repetition by calling a dedicated method and add it to the collection.
217  repetition = repeatNode.getExpressionGenerator( getCopiedTree( originalExpression), AND_OPERATOR).buildTree();
218  else { // create all the other repetitions
219  List< ExpressionNode<?>> leafNodes = new ArrayList< ExpressionNode<?>>();
220  // create the leaf node to be added when the binary tree is composed (in the above example for i=4: [E,E,E,E])
221  for( int j = 0; j < i; j++)
222  leafNodes.add( getCopiedTree( originalExpression));
223  // create the binary tree and add to the collection
224  repetition = repeatNode.getExpressionGenerator( leafNodes, AND_OPERATOR).buildTree();
225  }
226  // store the repetition node ranges to be used on evaluator
227  repetitionRanges.add( new RepeatitionRange( repetition, i));
228  // store the trees that represents the repetition to be added later
229  repetitionExpressions.add( repetition);
230  //repetition.showTree( "repetition tree for order: " + i);
231  }
232 
233  // create the binary tree (that include the previous one) to represent different repetition of the expression
234  // ex: repeat(0-0)=E(0) ; repeat(0-1)=E(0)|E(1) ; ... ; repeat(n-m)=E(n)|E(n+1)|...|E(m)
235  ExpressionNode<?> logicalExpression = repeatNode.getExpressionGenerator( repetitionExpressions, OR_OPERATOR).buildTree();
236  return logicalExpression;
237  }
